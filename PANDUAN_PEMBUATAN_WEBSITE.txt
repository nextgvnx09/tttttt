// components/Navbar.tsx
import React from 'react';
import { Page } from '../types';
import { Menu, X } from 'lucide-react';

interface NavbarProps {
  activePage: Page;
  setActivePage: (page: Page) => void;
}

const Navbar: React.FC<NavbarProps> = ({ activePage, setActivePage }) => {
  const [isOpen, setIsOpen] = React.useState(false);

  const navItems = [
    { page: Page.HOME, label: 'Beranda' },
    { page: Page.PROFIL, label: 'Profil' },
    { page: Page.GURU, label: 'Guru' },
    { page: Page.KEGIATAN, label: ' Kegiatan' },
    { page: Page.PRESTASI, label: 'Prestasi' },
    { page: Page.BERITA, label: 'Berita' },
    { page: Page.GALERI, label: 'Galeri' },
    { page: Page.PPDB, label: 'PPDB' },
  ];

  return (
    <nav className="bg-primary text-white sticky top-0 z-50">
      <div className="container mx-auto px-4">
        <div className="flex justify-between items-center h-16">
          <div className="text-xl font-bold text-accent">
            SMAN 2 Tompaso
          </div>
          
          {/* Desktop Menu */}
          <div className="hidden md:flex space-x-4">
            {navItems.map(item => (
              <button
                key={item.page}
                onClick={() => setActivePage(item.page)}
                className={`px-3 py-2 rounded-md transition-colors ${
                  activePage === item.page 
                    ? 'bg-accent text-primary' 
                    : 'hover:bg-white/10'
                }`}
              >
                {item.label}
              </button>
            ))}
            <button 
              onClick={() => setActivePage(Page.GURU_PORTAL)}
              className="px-4 py-2 bg-accent text-primary rounded-md font-medium"
            >
              GURU
            </button>
          </div>

          {/* Mobile Menu Button */}
          <button 
            className="md:hidden"
            onClick={() => setIsOpen(!isOpen)}
          >
            {isOpen ? <X /> : <Menu />}
          </button>
        </div>
      </div>
    </nav>
  );
};

export default Navbar;
```

// CRUD Operations for News
export function getNews(): NewsItem[] {
  const news = localStorage.getItem('news');
  return news ? JSON.parse(news) : [];
}

export function addNews(news: NewsItem): void {
  const allNews = getNews();
  allNews.unshift(news);
  localStorage.setItem('news', JSON.stringify(allNews));
}

export function updateNews(id: string | number, updatedNews: NewsItem): void {
  const allNews = getNews();
  const index = allNews.findIndex(n => n.id === id);
  if (index !== -1) {
    allNews[index] = updatedNews;
    localStorage.setItem('news', JSON.stringify(allNews));
  }
}

export function deleteNews(id: string | number): void {
  const allNews = getNews().filter(n => n.id !== id);
  localStorage.setItem('news', JSON.stringify(allNews));
}

### Langkah 7: Membuat Dashboard Admin

```
typescript
// components/AdminDashboard.tsx
import React, { useState } from 'react';
import { getNews, addNews, updateNews, deleteNews } from '../lib/actions';
import { NewsItem } from '../types';

const AdminDashboard: React.FC = () => {
  const [news, setNews] = useState<NewsItem[]>(getNews());
  const [showModal, setShowModal] = useState(false);
  const [editingNews, setEditingNews] = useState<NewsItem | null>(null);

  const handleSave = (newsItem: NewsItem) => {
    if (editingNews) {
      updateNews(newsItem.id, newsItem);
    } else {
      addNews({ ...newsItem, id: Date.now() });
    }
    setNews(getNews());
    setShowModal(false);
    setEditingNews(null);
  };

  const handleDelete = (id: string | number) => {
    if (confirm('Apakah Anda yakin ingin menghapus berita ini?')) {
      deleteNews(id);
      setNews(getNews());
    }
  };

  return (
    <div className="min-h-screen bg-gray-100 p-8">
      <div className="container mx-auto">
        <h1 className="text-3xl font-bold mb-8">Dashboard Admin</h1>
        
        {/* Stats Cards */}
        <div className="grid grid-cols-1 md:grid-cols-4 gap-4 mb-8">
          <div className="bg-white p-6 rounded-lg shadow">
            <h3 className="text-gray-500">Total Berita</h3>
            <p className="text-3xl font-bold">{news.length}</p>
          </div>
          {/* More stats... */}
        </div>

        {/* News Table */}
        <div className="bg-white rounded-lg shadow p-6">
          <div className="flex justify-between items-center mb-4">
            <h2 className="text-xl font-semibold">Kelola Berita</h2>
            <button 
              onClick={() => setShowModal(true)}
              className="bg-accent text-primary px-4 py-2 rounded-md"
            >
              + Tambah Berita
            </button>
          </div>
          
          <table className="w-full">
            <thead>
              <tr className="border-b">
                <th className="text-left py-2">Judul</th>
                <th className="text-left py-2">Kategori</th>
                <th className="text-left py-2">Tanggal</th>
                <th className="text-left py-2">Aksi</th>
              </tr>
            </thead>
            <tbody>
              {news.map(item => (
                <tr key={item.id} className="border-b">
                  <td className="py-2">{item.title}</td>
                  <td className="py-2">{item.category}</td>
                  <td className="py-2">{item.date}</td>
                  <td className="py-2">
                    <button 
                      onClick={() => { setEditingNews(item); setShowModal(true); }}
                      className="text-blue-600 mr-2"
                    >
                      Edit
                    </button>
                    <button 
                      onClick={() => handleDelete(item.id)}
                      className="text-red-600"
                    >
                      Hapus
                    </button>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};

export default AdminDashboard;
